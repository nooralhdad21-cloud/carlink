function calcGold(){

  document.getElementById('res-gold').classList.add('show');
  goldAI(k,w,type,craft,comm,gv,total);
}

function goldAI(k,w,type,craft,comm,gv,total){
  const tips=[];
  const normalCraft=4500;
  if(craft>0){
    const pct=(craft/GOLD[k]*100).toFixed(1);
    if(craft>7000) tips.push({t:'bad',m:`🚨 أجور الصياغة ${craft.toLocaleString()} د.ع/غ مرتفعة جداً. المعدل ببغداد 3,000–5,500 — جرب التفاوض.`});
    else if(craft>normalCraft) tips.push({t:'warn',m:`⚠️ أجور الصياغة مرتفعة قليلاً (${pct}% من قيمة الغرام). يمكن التفاوض.`});
    else tips.push({t:'ok',m:`✅ أجور الصياغة معقولة (${pct}%). ضمن النطاق الطبيعي.`});
  }
  if(comm>5) tips.push({t:'warn',m:`⚠️ عمولة ${comm}% مرتفعة. المكاتب الموثوقة تأخذ 1–3% كحد أقصى.`});
  if(type==='used') tips.push({t:'info',m:`💡 الذهب المستعمل خُصم 10%. إذا القطعة بحالة ممتازة جرب التفاوض على 5% فقط.`});
  if(type==='scrap') tips.push({t:'bad',m:`🔥 سعر الخردة يخصم 20%. تأكد من أنهم يحسبون فعلاً بسعر الذهب الحالي وليس سعر قديم.`});
  if(k===21) tips.push({t:'gold',m:`🟡 21 عيار الأكثر تداولاً. سعر البورصة اليوم ${fmt(GOLD[21])} دينار/غرام.`});

  setAI(tips);
}

const TO_USD={USD:1,EUR:1.095,GBP:1.27,SAR:0.2667,TRY:0.029,AED:0.2723,KWD:3.25,IRR:0.0000238,IQD:1/1310,CNY:0.1382,INR:0.012,JOD:1.41};
function calcCurr(){
  const from=document.getElementById('c-from').value;
  const amt=parseFloat(document.getElementById('c-amount').value)||0;
  const to=document.getElementById('c-to').value;
  document.getElementById('c-unit').textContent=from;
  if(amt<=0) return;
  const rate=selectedRate==='say'?R.say:selectedRate==='par'?R.par:R.off;
  const usd=amt*(TO_USD[from]||1);
  let res;
  if(to==='IQD') res=usd*rate;
  else if(from==='IQD') res=amt/rate/(TO_USD[to]||1);
  else res=usd/(TO_USD[to]||1);
  document.getElementById('rv-curr').innerHTML=fd(res,to)+' <span class="cur">'+to+'</span>';
  // triple comparison
  if(from!=='IQD'&&to==='IQD'){
    const offV=usd*R.off,sayV=usd*R.say,parV=usd*R.par;
    document.getElementById('tri-curr').innerHTML=
      `<div class="tri-c tri-off"><div class="tri-l">🏛️ رسمي</div><div class="tri-v">${fmt(offV)}</div><div class="tri-d">دينار</div></div>`+

      `<div class="tri-c tri-par"><div class="tri-l">🏪 موازي</div><div class="tri-v">${fmt(parV)}</div><div class="tri-d">دينار</div></div>`;
  } else document.getElementById('tri-curr').innerHTML='';
  document.getElementById('rb-curr').innerHTML=
    brow('المبلغ',amt.toLocaleString()+' '+from)+
    brow('السعر المستخدم',selectedRate==='say'?'صيرفة ('+R.say+')':selectedRate==='par'?'موازي ('+R.par+')':'رسمي ('+R.off+')')+
    (from!=='IQD'?brow('المعادل بالدولار',fd(usd,null)+' USD'):'')+
    browT(fd(res,to)+' '+to);
  document.getElementById('res-curr').classList.add('show');
  setAI([
    {t:'say',m:`🏧 الصيرفة تعطيك ${fmt(usd*R.say)} دينار لكل ${fmt(usd*100/100)} دولار — أفضل خيار متاح.`},

  ]);
}

function genD(base,days,vol){
  const d=[],l=[];let p=base;const now=new Date();
  for(let i=days;i>=0;i--){
    const dt=new Date(now);dt.setDate(dt.getDate()-i);

    p+=(Math.random()-.49)*vol;p=Math.max(base*.93,Math.min(base*1.07,p));

}
